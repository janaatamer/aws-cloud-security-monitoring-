import boto3
from datetime import datetime

sns_client = boto3.client("sns")
SNS_TOPIC_ARN = "arn:aws:sns:eu-north-1:369012867517:security-alerts"

def lambda_handler(event, context):

    detail = event.get("detail", {})

    user_type = detail.get("userIdentity", {}).get("type", "Unknown")
    source_ip = detail.get("sourceIPAddress", "Unknown")

    region = event.get("region", "Unknown")
    time = event.get("time", datetime.utcnow().isoformat())

    message = f"""
AWS Security Alert
=================
Event Type: Root Account Login

User Type: {user_type}
Source IP: {source_ip}
Region: {region}
Time: {time}

Action Required:
Investigate immediately if this login is unexpected.
"""

    sns_client.publish(
        TopicArn=SNS_TOPIC_ARN,
        Subject="CRITICAL: AWS Root Account Login Detected",
        Message=message
    )

    return {"statusCode": 200}
