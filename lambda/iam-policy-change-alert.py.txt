import boto3
from datetime import datetime

sns_client = boto3.client("sns")
SNS_TOPIC_ARN = "arn:aws:sns:eu-north-1:369012867517:security-alerts"

def lambda_handler(event, context):

    detail = event.get("detail", {})

    user = detail.get("userIdentity", {}).get("userName", "Unknown")
    event_name = detail.get("eventName", "Unknown")

    region = event.get("region", "Unknown")
    time = event.get("time", datetime.utcnow().isoformat())

    message = f"""
AWS Security Alert
=================
Event Type: IAM Policy Change

User: {user}
Action: {event_name}
Region: {region}
Time: {time}

Action Required:
Verify that IAM permission changes are authorized.
"""

    sns_client.publish(
        TopicArn=SNS_TOPIC_ARN,
        Subject="AWS Security Alert: IAM Policy Change",
        Message=message
    )

    return {"statusCode": 200}
